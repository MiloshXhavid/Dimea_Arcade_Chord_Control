struct ProcessParams
{
    double   sampleRate    = 44100.0;
    int      blockSize     = 512;
    double   bpm           = 120.0;        // effective BPM (free or DAW)

    // Sync mode
    bool     syncMode      = false;        // false=free Hz, true=beat subdivisions
    double   ppqPosition   = -1.0;        // -1 = not available / DAW not playing
    bool     isDawPlaying  = false;

    // Free mode
    float    rateHz        = 1.0f;        // 0.01..20.0 Hz (log scale)

    // Sync mode — subdivision in beats (quarter-note = 1.0 beat)
    // e.g. 1/8=0.5, 1/4=1.0, dotted 1/4=1.5, triplet 1/4=0.6667
    double   subdivBeats   = 1.0;

    // Maximum slow subdivision: loopLengthBars x beatsPerBar, passed from PluginProcessor
    double   maxCycleBeats = 16.0;

    // Waveform parameters
    Waveform waveform      = Waveform::Sine;
    float    phaseShift    = 0.0f;         // 0.0..1.0 maps 0..360 degrees
    float    distortion    = 0.0f;         // 0.0..1.0
    float    level         = 1.0f;         // 0.0..1.0 amplitude / depth
};

**process() implementation — follow this logic exactly:**

1. Clamp subdivBeats to maxCycleBeats: `const double cycleBeats = p.syncMode ? std::min(p.subdivBeats, p.maxCycleBeats) : 1.0;`

2. Compute cycleSeconds and cycleSamples for sync mode (DAW stopped):
```cpp
const double cycleSeconds = (p.syncMode ? cycleBeats : 1.0) * (60.0 / p.bpm);
const double cycleSamples = cycleSeconds * p.sampleRate;
```
For free mode: `const double freeModeCycleSamples = (1.0 / p.rateHz) * p.sampleRate;`

3. Transport restart detection — place BEFORE the phase derivation branches:
```cpp
if (!prevDawPlaying_ && p.isDawPlaying)
    sampleCount_ = 0;   // reset free/stopped counter on DAW restart
prevDawPlaying_ = p.isDawPlaying;
```

4. Phase derivation — three branches:

**Branch A: Sync + DAW playing** (`p.syncMode && p.isDawPlaying && p.ppqPosition >= 0.0`):
```cpp
normalizedPhase = std::fmod(p.ppqPosition, cycleBeats) / cycleBeats;
// Keep sampleCount_ in sync for seamless DAW-stop fallback:
sampleCount_ = static_cast<int64_t>(normalizedPhase * cycleSamples);
// S&H cycle index from ppqPos:
currentCycle = static_cast<int64_t>(p.ppqPosition / cycleBeats);
```

**Branch B: Sync + DAW stopped** (`p.syncMode && !p.isDawPlaying`):
```cpp
normalizedPhase = std::fmod(static_cast<double>(sampleCount_), cycleSamples) / cycleSamples;
sampleCount_ += p.blockSize;
currentCycle = static_cast<int64_t>(sampleCount_ / cycleSamples);
```

**Branch C: Free mode** (`!p.syncMode`):
```cpp
const double phaseIncrement = (static_cast<double>(p.rateHz) / p.sampleRate) * p.blockSize;
phase_ = std::fmod(phase_ + phaseIncrement, 1.0);
normalizedPhase = phase_;
totalCycles_ += phaseIncrement;
currentCycle  = static_cast<int64_t>(totalCycles_);
```

5. S&H boundary detection (applies to all branches):
```cpp
if (p.waveform == Waveform::SH)
{
    if (currentCycle != prevCycle_)
    {
        shHeld_    = nextLcg();   // new random value — instant jump, no interpolation
        prevCycle_ = currentCycle;
    }
}
```

6. Apply phaseShift AFTER computing normalizedPhase:
```cpp
normalizedPhase = std::fmod(normalizedPhase + static_cast<double>(p.phaseShift), 1.0);
if (normalizedPhase < 0.0) normalizedPhase += 1.0;  // guard against negative fmod
```

7. Evaluate waveform:
```cpp
float output = evaluateWaveform(p.waveform, static_cast<float>(normalizedPhase));
```

8. Distortion (additive LCG noise — skip for Random waveform per CONTEXT.md):
```cpp
if (p.waveform != Waveform::Random && p.distortion > 0.0f)
    output += p.distortion * nextLcg();
```

9. Apply level:
```cpp
output *= p.level;
return output;
```

**evaluateWaveform() implementation:**
```cpp
float LfoEngine::evaluateWaveform(Waveform w, float phi) const
{
    switch (w)
    {
        case Waveform::Sine:
            return std::sin(phi * kTwoPi);                        // [-1, +1]

        case Waveform::Triangle:
            // phi=0→-1, phi=0.25→+1, phi=0.5→-1, phi=0.75→+1
            return (phi < 0.5f) ? (4.0f * phi - 1.0f) : (3.0f - 4.0f * phi);

        case Waveform::SawUp:
            return 2.0f * phi - 1.0f;                             // -1→+1

        case Waveform::SawDown:
            return 1.0f - 2.0f * phi;                             // +1→-1

        case Waveform::Square:
            return (phi < 0.5f) ? 1.0f : -1.0f;

        case Waveform::SH:
            return shHeld_;                                        // set by S&H boundary detection above

        case Waveform::Random:
            return const_cast<LfoEngine*>(this)->nextLcg();       // per-sample noise, rate-agnostic
                                                                   // distortion is NOT applied on top (see CONTEXT.md)
        default:
            return 0.0f;
    }
}
```
