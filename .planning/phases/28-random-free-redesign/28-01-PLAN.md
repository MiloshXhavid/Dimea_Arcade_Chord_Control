**Step A: Add Poisson draw helper function** (file-scope static, immediately after `hitsPerBarToProbability` function):

```cpp
// Exponential inter-arrival time for Poisson process with rate λ = population events/bar.
// Uses existing LCG nextRandom() (audio-thread safe).
// freeTempoBpm = p.randomFreeTempo (NOT p.bpm — free mode always uses free tempo).
static float drawPoissonWaitSamples(float population, double sampleRate, double freeTempoBpm,
                                     uint32_t& rng)
{
    // samples per bar at 4/4 time
    const double samplesPerBar = (sampleRate * 60.0 / freeTempoBpm) * 4.0;
    const double meanWait      = samplesPerBar / static_cast<double>(std::max(1.0f, population));
    // Inverse-CDF exponential draw; guard U=0 to avoid log(0) = -inf
    const uint32_t raw = rng * 1664525u + 1013904223u;
    rng = raw;
    const float u = std::max(1e-6f, static_cast<float>(raw >> 1) / float(0x7FFFFFFF));
    const float wait = static_cast<float>(-meanWait * std::log(static_cast<double>(u)));
    // Apply 10ms minimum floor (matches minDurationSamples pattern in the file)
    const float minWait = static_cast<float>(0.010 * sampleRate);
    return std::max(minWait, wait);
}
```
