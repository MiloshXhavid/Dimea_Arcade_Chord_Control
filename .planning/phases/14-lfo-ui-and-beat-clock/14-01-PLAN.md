```cpp
// Beat clock detection — fires beatOccurred_ once per beat
{
    const int blockSize = buffer.getNumSamples();

    double currentBeatCount;
    if (isDawPlaying && ppqPos >= 0.0)
    {
        // DAW playing: ppqPos advances by BPM * blockSize / (60 * sampleRate) per block
        currentBeatCount = ppqPos;
    }
    else
    {
        // Free tempo: derive from sample counter
        const float bpm = effectiveBpm_.load(std::memory_order_relaxed);
        currentBeatCount = (static_cast<double>(bpm) / 60.0)
                         * static_cast<double>(sampleCounter_) / sampleRate_;
    }

    // Static local persists across processBlock calls (audio-thread only — no race)
    static double sBeatCountPrev = -1.0;
    if (sBeatCountPrev >= 0.0
        && std::floor(currentBeatCount) > std::floor(sBeatCountPrev))
    {
        beatOccurred_.store(true, std::memory_order_relaxed);
    }
    sBeatCountPrev = currentBeatCount;

    // Advance free-tempo sample counter (also used by LFO sync modes)
    sampleCounter_ += blockSize;
}
```
