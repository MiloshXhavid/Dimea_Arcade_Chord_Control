finaliseRecording() [private, audio-thread-only — PUNCH-IN MERGE LOGIC]:
  This is the critical method that implements the locked decision:
  "new events replace old at beat positions touched by the new pass; untouched positions are preserved."

  Implementation:
  const int newCount = fifo_.getNumReady();

  // Case 1: no existing content — simple drain of FIFO into playbackStore_
  if (playbackCount_.load(std::memory_order_relaxed) == 0 || newCount == 0) {
    int count = 0;
    if (newCount > 0) {
      auto scope = fifo_.read(newCount);
      scope.forEach([&](int idx) {
        if (count &lt; LOOPER_FIFO_CAPACITY)
          playbackStore_[count++] = eventBuf_[idx];
      });
      fifo_.reset();
    }
    playbackCount_.store(count, std::memory_order_release);
    return;
  }

  // Case 2: punch-in merge — existing content present AND new events recorded
  // Step A: drain new events from FIFO into a temporary local array
  LooperEvent newEvents[LOOPER_FIFO_CAPACITY];
  int newEventCount = 0;
  {
    auto scope = fifo_.read(newCount);
    scope.forEach([&](int idx) {
      if (newEventCount &lt; LOOPER_FIFO_CAPACITY)
        newEvents[newEventCount++] = eventBuf_[idx];
    });
  }
  fifo_.reset();

  // Step B: build a beat-window bitmask of positions touched by new pass.
  // "Touched" = within ±(loopLen / 64) of any newly recorded event's beatPosition.
  // This is approximately ±1/16 of a beat at loop resolution — sufficient for block-granular recording.
  const double loopLen = getLoopLengthBeats();
  const double touchRadius = (loopLen > 0.0) ? (loopLen / 64.0) : 0.05;

  // Step C: build merged result — old events that are NOT in any touched window, plus all new events.
  // Use a fixed stack array (no heap alloc on audio thread).
  LooperEvent merged[LOOPER_FIFO_CAPACITY];
  int mergedCount = 0;

  // First pass: keep old events that fall OUTSIDE all touched beat windows
  const int oldCount = playbackCount_.load(std::memory_order_relaxed);
  for (int i = 0; i &lt; oldCount &amp;&amp; mergedCount &lt; LOOPER_FIFO_CAPACITY; ++i) {
    const double oldPos = playbackStore_[i].beatPosition;
    bool inTouchedWindow = false;
    for (int j = 0; j &lt; newEventCount; ++j) {
      const double dist = std::abs(oldPos - newEvents[j].beatPosition);
      // Handle wrap-around at loop boundary
      const double distWrapped = std::min(dist, loopLen - dist);
      if (distWrapped &lt;= touchRadius) {
        inTouchedWindow = true;
        break;
      }
    }
    if (!inTouchedWindow)
      merged[mergedCount++] = playbackStore_[i];
  }

  // Second pass: append all new events
  for (int j = 0; j &lt; newEventCount &amp;&amp; mergedCount &lt; LOOPER_FIFO_CAPACITY; ++j)
    merged[mergedCount++] = newEvents[j];

  // Copy merged result back into playbackStore_
  for (int i = 0; i &lt; mergedCount; ++i)
    playbackStore_[i] = merged[i];
  playbackCount_.store(mergedCount, std::memory_order_release);

recordGate(double beatPos, int voice, bool on):
  ASSERT_AUDIO_THREAD();
  if (!recording_.load(std::memory_order_relaxed)) return;
  if (!recGates_.load(std::memory_order_relaxed)) return;
  const double loopLen = getLoopLengthBeats();
  if (loopLen &lt;= 0.0) return;
  // Cap check — auto-stop when loop fills
  if (fifo_.getFreeSpace() &lt; 1) {
    capReached_.store(true, std::memory_order_relaxed);
    recording_.store(false, std::memory_order_relaxed);
    return;
  }
  const double pos = std::fmod(beatPos, loopLen);
  LooperEvent e { pos, LooperEvent::Type::Gate, voice, on ? 1.0f : 0.0f };
  auto scope = fifo_.write(1);
  scope.forEach([&](int idx) { eventBuf_[idx] = e; });

recordJoystick(double beatPos, float x, float y):
  ASSERT_AUDIO_THREAD();
  if (!recording_.load(std::memory_order_relaxed)) return;
  if (!recJoy_.load(std::memory_order_relaxed)) return;
  const double loopLen = getLoopLengthBeats();
  if (loopLen &lt;= 0.0) return;
  const float thresh = joystickThresh_.load(std::memory_order_relaxed);
  const bool xMoved = std::abs(x - lastRecordedX_) > thresh;
  const bool yMoved = std::abs(y - lastRecordedY_) > thresh;
  if (!xMoved &amp;&amp; !yMoved) return;
  const double pos = std::fmod(beatPos, loopLen);
  if (xMoved) {
    if (fifo_.getFreeSpace() &lt; 1) { capReached_.store(true, std::memory_order_relaxed); return; }
    LooperEvent ex { pos, LooperEvent::Type::JoystickX, -1, x };
    auto scope = fifo_.write(1); scope.forEach([&](int idx){ eventBuf_[idx] = ex; });
    lastRecordedX_ = x;
  }
  if (yMoved) {
    if (fifo_.getFreeSpace() &lt; 1) { capReached_.store(true, std::memory_order_relaxed); return; }
    LooperEvent ey { pos, LooperEvent::Type::JoystickY, -1, y };
    auto scope = fifo_.write(1); scope.forEach([&](int idx){ eventBuf_[idx] = ey; });
    lastRecordedY_ = y;
  }

anchorToBar(double ppqPosition, double bpb):
  const double barIndex = std::floor(ppqPosition / bpb);
  loopStartPpq_ = barIndex * bpb;

process(const ProcessParams& p):
  ASSERT_AUDIO_THREAD();
  BlockOutput out{};

  // Service destructive requests first (safe: audio thread is sole FIFO user at this point)
  if (resetRequest_.exchange(false, std::memory_order_acq_rel)) {
    fifo_.reset();
    playbackStore_ = {};
    playbackCount_.store(0, std::memory_order_relaxed);
    internalBeat_  = 0.0;
    loopStartPpq_  = -1.0;
    playing_.store(false, std::memory_order_relaxed);
    recording_.store(false, std::memory_order_relaxed);
    playbackBeat_.store(0.0f, std::memory_order_relaxed);
    return out;
  }
  if (deleteRequest_.exchange(false, std::memory_order_acq_rel)) {
    fifo_.reset();
    playbackStore_ = {};
    playbackCount_.store(0, std::memory_order_relaxed);
    return out;
  }

  if (!playing_.load(std::memory_order_relaxed)) return out;

  const double loopLen = getLoopLengthBeats();
  if (loopLen &lt;= 0.0) return out;

  const double bpm = (p.bpm > 0.0) ? p.bpm : 120.0;
  const double samplesPerBeat = (p.sampleRate * 60.0) / bpm;

  // Determine current beat position within the loop
  double beatAtBlockStart;
  const bool useDaw = syncToDaw_.load(std::memory_order_relaxed)
                      &amp;&amp; p.isDawPlaying
                      &amp;&amp; p.ppqPosition >= 0.0;

  if (useDaw) {
    if (loopStartPpq_ &lt; 0.0)
      anchorToBar(p.ppqPosition, beatsPerBar());
    const double elapsed = p.ppqPosition - loopStartPpq_;
    beatAtBlockStart = std::fmod(elapsed &lt; 0.0 ? 0.0 : elapsed, loopLen);
  } else {
    if (!useDaw &amp;&amp; syncToDaw_.load(std::memory_order_relaxed)) {
      // DAW sync on but DAW not playing — pause internal advance
      playbackBeat_.store((float)std::fmod(internalBeat_, loopLen), std::memory_order_relaxed);
      return out;
    }
    loopStartPpq_ = -1.0;  // reset anchor when switching to internal clock
    beatAtBlockStart = std::fmod(internalBeat_, loopLen);
  }

  const double blockBeats   = static_cast&lt;double&gt;(p.blockSize) / samplesPerBeat;
  const double rawEnd       = beatAtBlockStart + blockBeats;
  const double beatAtEnd    = std::fmod(rawEnd, loopLen);
  const bool   wraps        = rawEnd >= loopLen;

  playbackBeat_.store((float)beatAtBlockStart, std::memory_order_relaxed);

  // Advance internal clock
  if (!useDaw) {
    internalBeat_ += blockBeats;
    if (internalBeat_ >= loopLen)
      internalBeat_ = std::fmod(internalBeat_, loopLen);
  }

  // Scan playbackStore_ (audio-thread-only during playback; recording_=false when playing_=true
  // in the sequential model — punch-in requires stop before re-record)
  const int count = playbackCount_.load(std::memory_order_acquire);
  for (int i = 0; i &lt; count; ++i) {
    const auto&amp; ev = playbackStore_[i];
    bool inWindow = wraps
        ? (ev.beatPosition >= beatAtBlockStart || ev.beatPosition &lt; beatAtEnd)
        : (ev.beatPosition >= beatAtBlockStart &amp;&amp; ev.beatPosition &lt; beatAtEnd);
    if (!inWindow) continue;

    switch (ev.type) {
      case LooperEvent::Type::JoystickX:
        out.hasJoystickX = true; out.joystickX = ev.value; break;
      case LooperEvent::Type::JoystickY:
        out.hasJoystickY = true; out.joystickY = ev.value; break;
      case LooperEvent::Type::Gate:
        if (ev.voice >= 0 &amp;&amp; ev.voice &lt; 4) {
          if (ev.value > 0.5f) out.gateOn[ev.voice] = true;
          else                  out.gateOff[ev.voice] = true;
        }
        break;
    }
  }

  return out;
